{%- comment -%} このファイルは、_dataファイルで定義されたタクソノミー情報と Jekyllの標準機能で取得した記事のカウント情報をマージし、 警告情報も付与したタクソノミーアイテムのリストを準備します。 HTML出力は行わず、Liquid変数を設定するのみです。 呼び出し元で以下の変数にアクセスできます: - merged_items: 処理済みタクソノミーアイテムの配列 必要な引数: - type: 'categories' または 'tags' (必須) - lang: 現在の言語 (必須) - warnings_data: 警告メッセージデータ (site.data.lang.taxonomy_warningなど) (必須) - schema: site.data.taxonomy.taxonomy_schema (必須) {%- endcomment -%} {%- assign type = include.type -%} {%- assign lang = include.lang -%} {%- assign warnings_data = include.warnings_data -%} {%- assign schema = include.schema -%} {# _dataファイルからタクソノミーの基本データを読み込む #} {# ファイルが存在しない場合にも空の配列が割り当てられるように default: "" | split: "" を追加 #} {%- assign data = site.data.taxonomy[type] | default: "" | split: "" -%} {# data が存在しない（空の配列になった）場合は、早期に処理を終了 #} {%- if data.size == 0 -%} {%- assign merged_items = "" | split: "" -%} {# 空のリストを返す #} {%- return -%} {%- endif -%} {# Jekyllの標準機能を使って、各タクソノミーに紐づく記事のカウントデータを取得 #} {%- if type == 'categories' -%} {%- assign standard_taxonomy_posts = site.categories -%} {%- elsif type == 'tags' -%} {%- assign standard_taxonomy_posts = site.tags -%} {%- else -%} {%- assign standard_taxonomy_posts = null -%} {%- endif -%} {%- assign seen_slugs = "" | split: "" -%} {%- assign merged_items = "" | split: "" -%} {# _dataのアイテムにカウントをマージし、警告情報を付与 #} {%- assign items = data | sort: 'taxonomy_priority' %} {%- for item in items %} {# _dataファイルから取得したタクソノミー情報を基に警告メッセージを生成 #} {%- assign current_item_warning = "" -%} {%- assign current_item_name = item.taxonomy_name[lang] | default: '無名' -%} {%- assign current_item_slug = item.taxonomy_slug | default: '' -%} {%- if current_item_name == '' %} {% assign current_item_warning = current_item_warning | append: "⚠️ " | append: warnings_data.name_missing[lang] | default: warnings_data.name_missing['ja'] | append: " " %} {%- endif %} {%- if current_item_slug == '' %} {% assign current_item_warning = current_item_warning | append: "⚠️ " | append: warnings_data.slug_missing[lang] | default: warnings_data.slug_missing['ja'] | append: " " %} {%- endif %} {%- if seen_slugs contains current_item_slug and current_item_slug != '' %} {% assign current_item_warning = current_item_warning | append: "⚠️ " | append: warnings_data.slug_duplicate[lang] | default: warnings_data.slug_duplicate['ja'] | append: " " %} {%- else %} {%- if current_item_slug != '' %}{% assign seen_slugs = seen_slugs | push: current_item_slug %}{% endif -%} {%- endif %} {# Jekyll標準のタクソノミーデータからカウント数を取得し、マージ #} {%- assign count = 0 -%} {%- for standard_name_posts_pair in standard_taxonomy_posts -%} {%- assign standard_name = standard_name_posts_pair[0] -%} {%- assign standard_posts = standard_name_posts_pair[1] -%} {%- if standard_name | slugify == current_item_slug -%} {%- assign count = standard_posts.size -%} {%- break -%} {# マッチしたらループを抜ける #} {%- endif --} {%- endfor -%} {# 新しいハッシュを作成して、必要な情報をまとめる #} {% assign merged_item = item | merge: {"_count": count, "_warning": current_item_warning} %} {% assign merged_items = merged_items | push: merged_item %} {%- endfor %}
